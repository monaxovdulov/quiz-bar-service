def main():
    start = time.perf_counter()

    topic = get_topic()
    questions_count = 10

    tries = 0
    questions = []

    while tries < 3:
        urls = get_urls_articles(topic, questions_count * 2)

        if not urls:
            tries += 1
            continue

        for url in urls:
            text = fetch_text_from(url)

            if not text:
                continue

            prompt = get_prompt(text)
            result = generate_text(prompt)

            if result:
                questions.append(result)

        if len(questions) >= questions_count:
            break

        tries += 1

    print(questions)
    print('\n\n')
    print(f"‚è± –í—Ä–µ–º—è –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è: {time.perf_counter() - start:.2f} —Å–µ–∫.")

import requests
from bs4 import BeautifulSoup
from .is_url_allowed import is_url_allowed


def fetch_text_from(url):
    if not is_url_allowed(url):
        return ""

    try:
        headers = {
            "User-Agent": (
                "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                "AppleWebKit/537.36 (KHTML, like Gecko) "
                "Chrome/121.0 Safari/537.36"
            ),
            "Accept-Language": "ru,en;q=0.9",
            "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8"
        }
        response = requests.get(url, headers=headers, timeout=1)
        soup = BeautifulSoup(response.text, "html.parser")

        for tag in soup(["script", "style", "noscript", "meta", "iframe"]):
            tag.extract()

        text = " ".join(soup.get_text().split())
        return text[:2500]
    except Exception:
        return ""

from ddgs import DDGS


def get_urls_articles(search_text, max_results=10):
    try:
        with DDGS() as ddgs:
            results = [r for r in ddgs.text(search_text, max_results=max_results)]

        return [r["href"] for r in results if "href" in r]
    except Exception:
        return None

from urllib.parse import urlparse

BLOCKED_DOMAINS = [
    "facebook.com",
    "youtube.com",
    "tiktok.com",
    "instagram.com",
    "reddit.com",
    "x.com",
    "vk.com",
    "ok.ru",
    "pinterest.com",
    "twitter.com",
]

PRIVATE_PREFIXES = ("127.", "192.168.", "localhost")

def is_url_allowed(url: str) -> bool:
    try:
        domain = urlparse(url).netloc.lower()

        if domain.startswith(PRIVATE_PREFIXES):
            return False

        return not any(bad in domain for bad in BLOCKED_DOMAINS)
    except Exception:
        return False
